#include <vector>
using namespace std;

vector<int> transformer_part3(vector<int> input, int hidden_dim) {
    vector<int> output;
    for (int i = 0; i < hidden_dim; i++) {
        int curr = input[i] * (1 / (1 + integer_exp(0 - input[i])));
        output.push_back(curr);
    }
    assert output == vec_elemwise_mul(vec_slice(input, 0, hidden_dim),scalar_vec_div(1,vec_scalar_add(1,vec_map(scalar_vec_sub(0, vec_slice(input, 0, hidden_dim)),lambda x: integer_exp(x)))))
;
}
