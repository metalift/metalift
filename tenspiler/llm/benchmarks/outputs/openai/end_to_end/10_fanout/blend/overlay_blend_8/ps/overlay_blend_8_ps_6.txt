
    def overlay_blend_8(base: List[List[int]], active: List[List[int]]) -> List[List[int]]:
        return matrix_where(
            base,
            active,
            lambda b, a: ite(
                b >= 16,
                2 * b + b - 2 * b * b // 32 - 32,
                2 * b * b // 32
            )
        )
