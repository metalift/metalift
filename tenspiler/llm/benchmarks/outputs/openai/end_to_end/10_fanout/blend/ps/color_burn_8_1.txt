
def color_burn_8(base: List[List[int]], active: List[List[int]]) -> List[List[int]]:
    return matrix_selection_two_args(
        matrix_scalar_add(255, matrix_elemwise_div(matrix_scalar_sub(255, base), active)),
        matrix_scalar_mul(255, matrix_elemwise_mul(active, active)),
        lambda x, y: ite(y != 0, x, y)
    )
