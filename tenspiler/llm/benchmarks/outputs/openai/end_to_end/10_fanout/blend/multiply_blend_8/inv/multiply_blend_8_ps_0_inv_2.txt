
from typing import List

def invariant1(base: List[List[int]], active: List[List[int]], out: List[List[int]], row: int) -> bool:
    return row >= 0 and row <= len(base) and out == matrix_elemwise_div(matrix_elemwise_mul(base[:row], active[:row]), matrix_scalar_mul(32, base[:row]))

def invariant2(base: List[List[int]], active: List[List[int]], out: List[List[int]], row: int, row_vec: List[int], col: int) -> bool:
    return col >= 0 and col <= len(base[0]) and row >= 0 and row <= len(base) and out == matrix_elemwise_div(matrix_elemwise_mul(base[:row], active[:row]), matrix_scalar_mul(32, base[:row])) and row_vec == matrix_elemwise_div(matrix_elemwise_mul(base[row][:col], active[row][:col]), vec_scalar_mul(32, base[row][:col]))
